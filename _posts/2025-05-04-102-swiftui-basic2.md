## 3: 상태 관리의 핵심

### 3.1 학습 목표
- @State의 동작 원리 완벽 이해
- @Binding으로 데이터 전달
- 단방향 데이터 흐름 이해

### 3.2 왜 @State가 필요한가?
```swift
// ❌ 작동하지 않음! (구조체는 불변)
struct CounterView: View {
    var count = 0  // 변경 불가능
    
    var body: some View {
        Button("Count: \(count)") {
            count += 1  // 컴파일 에러!
        }
    }
}

// ✅ @State 사용
struct CounterView: View {
    @State private var count = 0  // 변경 가능한 상태
    
    var body: some View {
        Button("Count: \(count)") {
            count += 1  // 작동! 화면도 자동 업데이트
        }
    }
}
```

**핵심 포인트**:
- @State는 SwiftUI가 관리하는 특별한 저장소
- 값이 변경되면 View가 자동으로 다시 그려짐
- `private`로 선언 (해당 View에서만 사용)

### 3.3 @Binding으로 데이터 공유
```swift
// 부모 View
struct ParentView: View {
    @State private var isOn = false
    
    var body: some View {
        VStack {
            Text("스위치: \(isOn ? "ON" : "OFF")")
            
            // 자식 View에 바인딩 전달 ($붙임!)
            ToggleView(isOn: $isOn)
        }
    }
}

// 자식 View
struct ToggleView: View {
    @Binding var isOn: Bool  // 부모의 데이터를 참조
    
    var body: some View {
        Toggle("토글", isOn: $isOn)
    }
}
```

**핵심 포인트**:
- @Binding은 다른 곳의 @State를 참조
- `$`는 Binding을 생성하는 연산자
- 양방향 데이터 바인딩

### 3.4 State vs Binding 비교표

| 구분 | @State | @Binding |
|------|--------|----------|
| **역할** | 데이터 소유 | 데이터 참조 |
| **선언** | `@State private var name = ""` | `@Binding var name: String` |
| **전달** | `ChildView(name: $name)` | 받기만 함 |
| **사용처** | 해당 View 내부 | 부모-자식 간 공유 |

1\.실습: 할 일 추가 앱
```swift
// 할 일 모델
struct Todo: Identifiable {
    let id = UUID()
    var title: String
    var isCompleted: Bool = false
}

// 메인 화면
struct TodoListView: View {
    @State private var todos: [Todo] = []
    @State private var newTodoTitle = ""
    
    var body: some View {
        VStack {
            // 입력 필드
            HStack {
                TextField("새 할 일", text: $newTodoTitle)
                    .textFieldStyle(.roundedBorder)
                
                Button("추가") {
                    addTodo()
                }
                .disabled(newTodoTitle.isEmpty)
            }
            .padding()
            
            // 할 일 목록
            List {
                ForEach($todos) { $todo in
                    TodoRow(todo: $todo)
                }
            }
        }
    }
    
    func addTodo() {
        let todo = Todo(title: newTodoTitle)
        todos.append(todo)
        newTodoTitle = ""
    }
}

// 할 일 행 컴포넌트
struct TodoRow: View {
    @Binding var todo: Todo
    
    var body: some View {
        HStack {
            Image(systemName: todo.isCompleted ? "checkmark.circle.fill" : "circle")
                .foregroundStyle(todo.isCompleted ? .green : .gray)
                .onTapGesture {
                    todo.isCompleted.toggle()
                }
            
            Text(todo.title)
                .strikethrough(todo.isCompleted)
                .foregroundStyle(todo.isCompleted ? .secondary : .primary)
        }
    }
}
```

struct ParentView: View {
    @State private var temperature: Double = 20
    @State private var isMetric = true
    
    var displayTemperature: String {
        if isMetric {
            return String(format: "%.1f°C", temperature)
        } else {
            return String(format: "%.1f°F", temperature * 9/5 + 32)
        }
    }
    
    var body: some View {
        VStack(spacing: 30) {
            Text("온도 조절기")
                .font(.title)
                .bold()
            
            Text(displayTemperature)
                .font(.system(size: 60, weight: .bold))
                .foregroundStyle(.blue)
            
            // 자식 View에 Binding 전달
            TemperatureSlider(temperature: $temperature)
            
            UnitToggle(isMetric: $isMetric)
            
            Text("부모 View가 온도(\(String(format: "%.1f", temperature))°C)와 단위(섭씨: \(isMetric ? "O" : "X"))를 관리합니다")
                .font(.caption)
                .foregroundStyle(.secondary)
                .multilineTextAlignment(.center)
        }
        .padding()
    }
}

// MARK: - 5. 할 일 앱 (완전한 예시)
struct Todo: Identifiable {
    let id = UUID()
    var title: String
    var isCompleted: Bool = false
    var priority: Priority = .normal
    
    enum Priority: String, CaseIterable {
        case low = "낮음"
        case normal = "보통"
        case high = "높음"
        
        var color: Color {
            switch self {
            case .low: return .green
            case .normal: return .blue
            case .high: return .red
            }
        }
    }
}

struct TodoListView: View {
    @State private var todos: [Todo] = [
        Todo(title: "SwiftUI 공부하기", priority: .high),
        Todo(title: "장보기", isCompleted: true, priority: .normal),
        Todo(title: "운동하기", priority: .low)
    ]
    @State private var newTodoTitle = ""
    @State private var selectedPriority: Todo.Priority = .normal
    @State private var showCompletedOnly = false
    
    var filteredTodos: [Todo] {
        if showCompletedOnly {
            return todos.filter { $0.isCompleted }
        }
        return todos
    }
    
    var body: some View {
        VStack(spacing: 0) {
            // 헤더
            VStack(spacing: 16) {
                Text("할 일 목록")
                    .font(.title)
                    .bold()
                
                // 통계
                HStack(spacing: 20) {
                    StatBadge(
                        count: todos.count,
                        label: "전체",
                        color: .blue
                    )
                    StatBadge(
                        count: todos.filter { $0.isCompleted }.count,
                        label: "완료",
                        color: .green
                    )
                    StatBadge(
                        count: todos.filter { !$0.isCompleted }.count,
                        label: "미완료",
                        color: .orange
                    )
                }
                
                // 입력 영역
                VStack(spacing: 12) {
                    HStack {
                        TextField("새 할 일", text: $newTodoTitle)
                            .textFieldStyle(.roundedBorder)
                        
                        Menu {
                            ForEach(Todo.Priority.allCases, id: \.self) { priority in
                                Button {
                                    selectedPriority = priority
                                } label: {
                                    Label(priority.rawValue, systemImage: selectedPriority == priority ? "checkmark" : "")
                                }
                            }
                        } label: {
                            Image(systemName: "flag.fill")
                                .foregroundStyle(selectedPriority.color)
                                .frame(width: 44, height: 44)
                                .background(.gray.opacity(0.1))
                                .clipShape(RoundedRectangle(cornerRadius: 8))
                        }
                        
                        Button {
                            addTodo()
                        } label: {
                            Image(systemName: "plus.circle.fill")
                                .font(.title2)
                        }
                        .disabled(newTodoTitle.isEmpty)
                    }
                    
                    Toggle("완료된 항목만 보기", isOn: $showCompletedOnly)
                        .font(.caption)
                }
            }
            .padding()
            .background(.gray.opacity(0.05))
            
            // 목록
            if filteredTodos.isEmpty {
                Spacer()
                VStack(spacing: 12) {
                    Image(systemName: "checkmark.circle")
                        .font(.system(size: 60))
                        .foregroundStyle(.gray)
                    Text(showCompletedOnly ? "완료된 할 일이 없습니다" : "할 일을 추가해보세요")
                        .foregroundStyle(.secondary)
                }
                Spacer()
            } else {
                List {
                    ForEach($filteredTodos) { $todo in
                        TodoRow(todo: $todo)
                    }
                    .onDelete(perform: deleteTodos)
                }
                .listStyle(.plain)
            }
        }
    }
    
    func addTodo() {
        let todo = Todo(title: newTodoTitle, priority: selectedPriority)
        todos.insert(todo, at: 0)
        newTodoTitle = ""
        selectedPriority = .normal
    }
    
    func deleteTodos(at offsets: IndexSet) {
        todos.remove(atOffsets: offsets)
    }
}

struct TodoRow: View {
    @Binding var todo: Todo
    
    var body: some View {
        HStack(spacing: 12) {
            // 완료 체크
            Button {

            } label: {
                Image(systemName: todo.isCompleted ? "checkmark.circle.fill" : "circle")
                    .font(.title3)
                    .foregroundStyle(todo.isCompleted ? .green : .gray)
            }
            .buttonStyle(.plain)
            
            // 제목
            VStack(alignment: .leading, spacing: 4) {
                Text(todo.title)
                    .strikethrough(todo.isCompleted)
                    .foregroundStyle(todo.isCompleted ? .secondary : .primary)
                
                HStack(spacing: 4) {
                    Image(systemName: "flag.fill")
                        .font(.caption2)
                    Text(todo.priority.rawValue)
                        .font(.caption)
                }
                .foregroundStyle(todo.priority.color)
            }
            
            Spacer()
        }
        .padding(.vertical, 8)
    }
}

## 4: List, Form, Navigation 마스터

### 4.1 학습 목표
- List와 ForEach의 차이 이해
- Form으로 입력 화면 구성
- NavigationStack 완벽 활용


### 4.2 List vs ForEach
```swift
// List: 스크롤 가능한 목록 (자동 스타일링)
List {
    Text("항목 1")
    Text("항목 2")
}

// ForEach: 반복문 (스타일 없음)
VStack {
    ForEach(items) { item in
        Text(item.name)
    }
}

// 조합: List + ForEach
List {
    ForEach(items) { item in
        Text(item.name)
    }
}
```

    let id = UUID()
    var name: String
    var phone: String
    var email: String
    var isFavorite: Bool = false

struct ContactListView: View {
    @State private var contacts = [
        Contact(name: "김철수", phone: "010-1234-5678", email: "chulsoo@email.com"),
        Contact(name: "이영희", phone: "010-8765-4321", email: "younghee@email.com", isFavorite: true)
    ]
    
    var body: some View {
        NavigationStack {
            List {
                Section("즐겨찾기") {
                    ForEach(contacts.filter { $0.isFavorite }) { contact in
                        NavigationLink {
                            ContactDetailView(contact: contact)
                        } label: {
                            ContactRow(contact: contact)
                        }
                    }
                }
                
                Section("모든 연락처") {
                    ForEach(contacts) { contact in
                        NavigationLink {
                            ContactDetailView(contact: contact)
                        } label: {
                            ContactRow(contact: contact)
                        }
                    }
                }
            }
            .navigationTitle("연락처")
        }
    }
}
