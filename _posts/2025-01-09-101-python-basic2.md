## 1. 실습 예제(로또 번호 생성기)


```python

import random

def generate_lotto():
    numbers = []
    while len(numbers) < 6:
        number = random.randint(1,45)
        if number not in numbers:
            numbers.append(number)
    return numbers 


def generate_lotto2():
    numbers = set()
    while len(numbers) < 6:
        number = random.randint(1, 45)
        numbers.add(number)
    return list(numbers)

def generate_lotto3():
    numbers = random.sample(range(1,46), 6)
    return numbers 

if __name__ == "__main__":
    for _ in range(3):
        lotto_number = generate_lotto3()
        print(sorted(lotto_number))

```

## 2. 실습 예제(번호 맞추기)

```python
import random

def guess_number():
    given_number = random.randint(1,100)
    guess_count = 0

    while True:
        try:
            guess = int(input("1부터 100 사이의 숫자를 입력하세요:"))
        except ValueError:
            print("숫자를 입력하세요")
            continue

        guess_count = guess_count + 1
        if guess < given_number:
            print("더 큰수를 입력하세요")
        elif guess > given_number:
            print("더 작은수를 입력하세요")
        else:
            print(f"축하합니다. {guess_count}만에 정답을 맞췄네요.")
            
            break

if __name__ == "__main__":
    guess_number()

```

## 3. 실습 예제(퀴즈 문제 풀기)

[quizzes.json](/assets/data/quizzes.json) 

```python
from dataclasses import dataclass 
import json

@dataclass
class Quiz:
    question:str
    options:list
    answer:str

def load_json(file_path="quizzes.json"):
    try:
        with open(file_path, "r") as f:
            quiz_data = json.load(f)
            return [Quiz(q["question"], q["options"],q["answer"]) for q in quiz_data]
    except FileNotFoundError:
        print(f"파일 {file_path}를 찾을 수 없습니다.")
        return []

def start_quiz():
    quizzes = load_json()
    if not quizzes:
        return 
    
    total_questions = len(quizzes)
    correct_answers = 0

    for i, quiz in enumerate(quizzes):
        print(f"Q{i+1}. {quiz.question}")
        for option in quiz.options:
            print(option)
        
        user_answer = input("Select (A, B, C, D) :").upper().strip()

        if user_answer == quiz.answer:
            print("Correct")
            correct_answers += 1
        else:
            print("Incorrect")

    score = (correct_answers / total_questions) * 100
    print(f"Your score: {score:.2f}%")

if __name__ == "__main__":
    start_quiz()

```

## 4. 실습 예제(Gemini API 활용하기)
- 환경 변수 파일 생성하기(.env)
    - 외부에 노출되면 안되는 API KEY 값들을 기재
- 필요한 라이브러리 설치(google-generativeai , dotenv)
    - `pip install google_generativeai dotenv`

```python
import google.generativeai as genai
import os
from dotenv import load_dotenv

# .env 파일 로드
load_dotenv()

# API 키 설정
genai.configure(api_key=os.getenv("GOOGLE_API_KEY"))

# 모델 초기화
model = genai.GenerativeModel("gemini-2.0-flash-exp")

# 콘텐츠 생성
response = model.generate_content("Explain how AI works in a few words")
print(response.text)
```

## 5. 실습 예제(OpenAI API 활용하기)
```python
from openai import OpenAI
from dotenv import load_dotenv

# .env 파일 로드
load_dotenv()

client = OpenAI() # 이제 환경변수에서 API 키를 자동으로 읽어옴
completion = client.chat.completions.create(
    model = "gpt-4",
    messages = [
        {
        "role":"user",
        "content":"Write a one-sentence bedtime story about a unicorn."
    }
    ]
)

print(completion.choices[0].message.content)
```
